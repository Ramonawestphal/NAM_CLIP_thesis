# HAM10000 Concept Prompts - VERSION 2
# Changes from v1:
#   - symmetric_uniform: replaced with "benign melanocytic nevus" framing
#     to avoid generic "mole" / "uniform brown" vocabulary that CLIP
#     matched on too aggressively against melanoma images.
#   - reticular_network: replaced with "delicate fine pigment grid" framing
#     to give CLIP visual qualifiers it can distinguish from "atypical".
# All other concepts unchanged from v1.
#
# Format: each concept has an ID, a short name, a tier, and 3 phrasing variants.
# Tier 1 = ABCD rule (Stolz et al., 1994)
# Tier 2 = 7-point checklist (Argenziano et al., 1998)
# Tier 3 = LLM-augmented class-discriminative concepts
# Template 1 = "dermoscopic image" (technical, CLIP-style)
# Template 2 = "dermatoscopic image" (alternative technical phrasing)
# Template 3 = "close-up photograph" (plain English, general CLIP)
#
# Total: 24 concepts x 3 phrasings = 72 prompts

# ============================================================
# TIER 1: ABCD RULE OF DERMATOSCOPY
# ============================================================

[asymmetry]
tier: 1
t1: a dermoscopic image of an asymmetric skin lesion
t2: a dermatoscopic image showing a skin lesion with asymmetric shape along both axes
t3: a close-up photograph of a mole that is asymmetric in outline

[border_irregularity]
tier: 1
t1: a dermoscopic image with an irregular and asymmetric border
t2: a dermatoscopic image showing a skin lesion with jagged, notched borders
t3: a close-up photograph of a mole with poorly defined edges

[colour_variation]
tier: 1
t1: a dermoscopic image of a lesion with multiple colours
t2: a dermatoscopic image showing variegated pigmentation including brown, black, and red
t3: a close-up photograph of a skin lesion with uneven color distribution

[diameter_large]
tier: 1
t1: a dermoscopic image of a large skin lesion greater than six millimetres
t2: a dermatoscopic image of a wide pigmented lesion
t3: a close-up photograph of a sizeable mole

# CHANGED in v2: was "symmetric mole with uniform brown colour" framing,
# which overlapped too heavily with generic pigmented-lesion vocabulary
# and scored high on melanoma. Now anchored on "melanocytic nevus".
[symmetric_uniform]
tier: 3
t1: a dermoscopic image of a benign melanocytic nevus
t2: a dermatoscopic image of a common acquired nevus
t3: a close-up photograph of a typical harmless mole
